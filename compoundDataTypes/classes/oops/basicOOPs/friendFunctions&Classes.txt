#include <iostream>
Using namespace std;

class Class
{
	int i;
public:
	Class()
	{
		i = 1;
	}
	friend void print (const Class &obj2);	/* Friend Function. Declared here, defined outside the class. It can be declared under any access 													   specifier. Friend Function can access members belonging to any Access Specifiers. It's different than a Method in 
									   the sense that it still requires a class type parameter to work. */

	friend void show(const Class &obj3)	// Friend Function defined inside the class.									
	{
		cout << obj3.i << endl;
	}
}:

void print(const Class &obj2)
{
	cout << obj2.i << endl;
}


int main()
{
	Class obj;
	print(obj);
	show(obj);
	return 0;
}
