	}
};

{
public:
	Derived1(  )
	{
		cout << "Derived1" << endl;
	}	
};

class Derived2 : virtual public Base
{
public:

	}	
};

class Derived3 : public Derived1, Derived2
{
public:
	Derived3(  )
	{
		cout << "Derived3" << endl;
	}	
};
